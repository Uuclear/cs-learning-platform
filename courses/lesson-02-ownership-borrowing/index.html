жүҖжңүжқғдёҺеҖҹз”ЁпјҡRustзҡ„ж ёеҝғйӯ”жі•

ејҖеңәзҷҪпјҡеӣҫд№ҰйҰҶзҡ„еҖҹд№Ұи§„еҲҷ

жғіиұЎдёҖдёӢдҪ еҺ»еӣҫд№ҰйҰҶеҖҹд№ҰгҖӮеӣҫд№ҰйҰҶжңүдёҖеҘ—дёҘж јзҡ„и§„еҲҷпјҡ

  1. жҜҸжң¬д№ҰеҸӘиғҪжңүдёҖдёӘеҖҹйҳ…иҖ…пјҲдёҚиғҪеҗҢж—¶еҖҹз»ҷеӨҡдёӘдәәпјү
  2. еҪ“дҪ еҪ’иҝҳд№ҰзұҚж—¶пјҢиҝҷжң¬д№Ұе°ұдёҚеҶҚеұһдәҺдҪ 
  3. дҪ еҸҜд»ҘжҠҠд№ҰеҖҹз»ҷеҲ«дәәзңӢпјҢдҪҶжңүдёҘж јйҷҗеҲ¶

Rust зҡ„жүҖжңүжқғзі»з»ҹе°ұеғҸиҝҷдёӘеӣҫд№ҰйҰҶз®ЎзҗҶзі»з»ҹпјҒе®ғеңЁзј–иҜ‘ж—¶е°ұзЎ®дҝқеҶ…еӯҳе®үе…ЁпјҢйҒҝе…ҚдәҶе…¶д»–иҜӯиЁҖдёӯеёёи§Ғзҡ„еҶ…еӯҳй”ҷиҜҜпјҢжҜ”еҰӮз©әжҢҮй’ҲгҖҒеҸҢйҮҚйҮҠж”ҫгҖҒж•°жҚ®з«һдәүзӯүгҖӮд»ҠеӨ©пјҢжҲ‘д»¬е°ұжқҘжҸӯејҖ Rust иҝҷдёӘ"ж ёеҝғйӯ”жі•"зҡ„зҘһз§ҳйқўзәұгҖӮ

ж ёеҝғжҰӮеҝөпјҡжүҖжңүжқғдёүи§„еҲҷ

Rust зҡ„жүҖжңүжқғзі»з»ҹе»әз«ӢеңЁдёүжқЎз®ҖеҚ•дҪҶејәеӨ§зҡ„и§„еҲҷд№ӢдёҠпјҡ

и§„еҲҷдёҖпјҡжҜҸдёӘеҖјйғҪжңүдёҖдёӘжүҖжңүиҖ…

еңЁ Rust дёӯпјҢжҜҸдёӘеҲҶй…ҚеңЁе ҶдёҠзҡ„еҖјйғҪжңүдё”д»…жңүдёҖдёӘеҸҳйҮҸдҪңдёәе…¶жүҖжңүиҖ…гҖӮиҝҷе°ұеғҸеӣҫд№ҰйҰҶзҡ„жҜҸжң¬д№ҰйғҪжңүе”ҜдёҖзҡ„еҖҹйҳ…и®°еҪ•гҖӮ

let s = String::from("hello"); // s жҳҜиҝҷдёӘеӯ—з¬ҰдёІзҡ„жүҖжңүиҖ…

и§„еҲҷдәҢпјҡдёҖж¬ЎеҸӘиғҪжңүдёҖдёӘжүҖжңүиҖ…

еҪ“жүҖжңүиҖ…зҰ»ејҖдҪңз”Ёеҹҹж—¶пјҢеҖјдјҡиў«иҮӘеҠЁжё…зҗҶпјҲйҖҡиҝҮ drop еҮҪж•°пјүгҖӮжӣҙйҮҚиҰҒзҡ„жҳҜпјҢеҪ“дҪ е°ҶеҖјиөӢз»ҷеҸҰдёҖдёӘеҸҳйҮҸж—¶пјҢжүҖжңүжқғдјҡеҸ‘з”ҹз§»еҠЁпјҲmoveпјүпјҢиҖҢдёҚжҳҜеӨҚеҲ¶гҖӮ

let s1 = String::from("hello");
let s2 = s1; // s1 зҡ„жүҖжңүжқғз§»еҠЁеҲ° s2
// println!("{}", s1); // й”ҷиҜҜпјҒs1 дёҚеҶҚжңүж•Ҳ

и§„еҲҷдёүпјҡеј•з”Ёеҝ…йЎ»жҖ»жҳҜжңүж•Ҳзҡ„

Rust дёҚе…Ғи®ёжӮ¬еһӮеј•з”ЁпјҲdangling referencesпјүвҖ”вҖ”жҢҮеҗ‘е·Із»Ҹиў«йҮҠж”ҫзҡ„еҶ…еӯҳзҡ„жҢҮй’ҲгҖӮзј–иҜ‘еҷЁдјҡзЎ®дҝқеј•з”ЁеңЁе…¶жҢҮеҗ‘зҡ„ж•°жҚ®еӯҳеңЁжңҹй—ҙе§Ӣз»Ҳжңүж•ҲгҖӮ

д»Јз Ғжј”зӨәпјҡRust жүҖжңүжқғзӨәдҫӢ

и®©жҲ‘д»¬йҖҡиҝҮеҮ дёӘе®һйҷ…дҫӢеӯҗжқҘзңӢзңӢжүҖжңүжқғжҳҜеҰӮдҪ•е·ҘдҪңзҡ„пјҡ

зӨәдҫӢ 1пјҡеҹәжң¬зҡ„жүҖжңүжқғиҪ¬з§»

fn main() {
    let s1 = String::from("hello");
    let s2 = s1; // жүҖжңүжқғд»Һ s1 з§»еҠЁеҲ° s2
    
    println!("{}", s2); // жӯЈеёёе·ҘдҪң
    // println!("{}", s1); // зј–иҜ‘й”ҷиҜҜпјҡs1 е·Із»Ҹиў«з§»еҠЁдәҶпјҒ
}

зӨәдҫӢ 2пјҡеҮҪж•°и°ғз”Ёдёӯзҡ„жүҖжңүжқғ

fn takes_ownership(some_string: String) {
    println!("{}", some_string);
} // some_string зҰ»ејҖдҪңз”ЁеҹҹпјҢи°ғз”Ё drop

fn main() {
    let s = String::from("hello");
    takes_ownership(s); // s зҡ„жүҖжңүжқғз§»еҠЁеҲ°еҮҪж•°
    // println!("{}", s); // й”ҷиҜҜпјҒs е·Із»Ҹиў«з§»еҠЁдәҶ
}

Copy Traitпјҡж ҲдёҠж•°жҚ®зҡ„дҫӢеӨ–

еҜ№дәҺз®ҖеҚ•зҡ„ж ҮйҮҸзұ»еһӢпјҲеҰӮж•ҙж•°гҖҒеёғе°”еҖјгҖҒжө®зӮ№ж•°зӯүпјүпјҢRust е®һзҺ°дәҶ Copy traitпјҢиҝҷж„Ҹе‘ізқҖе®ғд»¬еңЁиөӢеҖјж—¶дјҡиў«еӨҚеҲ¶иҖҢдёҚжҳҜз§»еҠЁпјҡ

let x = 5;
let y = x; // x иў«еӨҚеҲ¶пјҢдёҚжҳҜз§»еҠЁ
println!("x = {}, y = {}", x, y); // йғҪиғҪжӯЈеёёе·ҘдҪңпјҒ

ж·ұе…ҘзҗҶи§ЈпјҡеҖҹз”ЁжЈҖжҹҘеҷЁзҡ„е·ҘдҪңеҺҹзҗҶ

Rust зҡ„еҖҹз”ЁжЈҖжҹҘеҷЁпјҲborrow checkerпјүжҳҜзј–иҜ‘еҷЁзҡ„дёҖйғЁеҲҶпјҢе®ғеңЁзј–иҜ‘ж—¶еҲҶжһҗд»Јз Ғдёӯзҡ„еј•з”ЁпјҢзЎ®дҝқе®ғд»¬йҒөеҫӘд»ҘдёӢи§„еҲҷпјҡ

дёҚеҸҜеҸҳеҖҹз”Ёи§„еҲҷ

  • еңЁеҗҢдёҖж—¶й—ҙпјҢдҪ еҸҜд»Ҙжңүд»»ж„Ҹж•°йҮҸзҡ„дёҚеҸҜеҸҳеј•з”ЁпјҲ&Tпјү
  • дҪҶдёҚиғҪеҗҢж—¶жңүеҸҜеҸҳеј•з”Ё

еҸҜеҸҳеҖҹз”Ёи§„еҲҷ

  • еңЁеҗҢдёҖж—¶й—ҙпјҢдҪ еҸӘиғҪжңүдёҖдёӘеҸҜеҸҳеј•з”ЁпјҲ&mut Tпјү
  • еҪ“еӯҳеңЁеҸҜеҸҳеј•з”Ёж—¶пјҢдёҚиғҪжңүд»»дҪ•дёҚеҸҜеҸҳеј•з”Ё

иҝҷдәӣи§„еҲҷеҗ¬иө·жқҘеҫҲдёҘж јпјҢдҪҶе®ғд»¬жӯЈжҳҜ Rust иғҪеӨҹеңЁзј–иҜ‘ж—¶йҳІжӯўж•°жҚ®з«һдәүзҡ„е…ій”®пјҒ

let mut s = String::from("hello");

let r1 = &s; // дёҚеҸҜеҸҳеј•з”Ё
let r2 = &s; // еҸҰдёҖдёӘдёҚеҸҜеҸҳеј•з”Ё - OKпјҒ
println!("{} and {}", r1, r2);

let r3 = &mut s; // еҸҜеҸҳеј•з”Ё - ERRORпјҒ
// println!("{}, {}, and {}", r1, r2, r3);

е®һйҷ…еә”з”Ёпјҡеӯ—з¬ҰдёІеӨ„зҗҶдёҺж•°жҚ®з»“жһ„и®ҫи®Ў

еңЁе®һйҷ…зј–зЁӢдёӯпјҢжүҖжңүжқғе’ҢеҖҹз”Ёзі»з»ҹеҰӮдҪ•её®еҠ©жҲ‘д»¬еҶҷеҮәжӣҙеҘҪзҡ„д»Јз Ғпјҹ

еӯ—з¬ҰдёІеҲҮзүҮпјҲString Slicesпјү

еӯ—з¬ҰдёІеҲҮзүҮпјҲ&strпјүжҳҜеҜ№еӯ—з¬ҰдёІзҡ„дёҖйғЁеҲҶзҡ„еј•з”ЁпјҢе®ғд»¬дёҚжӢҘжңүж•°жҚ®пјҢеҸӘжҳҜеҖҹз”Ёпјҡ

fn first_word(s: &String) -> &str {
    let bytes = s.as_bytes();
    
    for (i, &item) in bytes.iter().enumerate() {
        if item == b' ' {
            return &s[0..i];
        }
    }
    
    &s[..]
}

жіЁж„ҸпјҡиҝҷдёӘеҮҪж•°жҺҘеҸ— &String иҖҢдёҚжҳҜ StringпјҢжүҖд»ҘдёҚдјҡиҺ·еҸ–жүҖжңүжқғпјҢи°ғз”ЁиҖ…д»Қз„¶еҸҜд»ҘдҪҝз”ЁеҺҹе§Ӣеӯ—з¬ҰдёІгҖӮ

ж•°жҚ®з»“жһ„дёӯзҡ„жүҖжңүжқғ

еңЁи®ҫи®Ўж•°жҚ®з»“жһ„ж—¶пјҢжҲ‘д»¬йңҖиҰҒд»”з»ҶиҖғиҷ‘жүҖжңүжқғпјҡ

struct User {
    username: String,
    email: String,
    sign_in_count: u64,
    active: bool,
}

fn build_user(email: String, username: String) -> User {
    User {
        email,        // жүҖжңүжқғз§»еҠЁиҝӣжқҘ
        username,     // жүҖжңүжқғз§»еҠЁиҝӣжқҘ  
        active: true,
        sign_in_count: 1,
    }
}

еёёи§ҒиҜҜеҢәпјҡиҜ•еӣҫеҗҢж—¶жӢҘжңүеӨҡдёӘеҸҜеҸҳеј•з”Ё

ж–°жүӢз»ҸеёёзҠҜзҡ„дёҖдёӘй”ҷиҜҜжҳҜиҜ•еӣҫеҗҢж—¶еҲӣе»әеӨҡдёӘеҸҜеҸҳеј•з”Ёпјҡ

let mut s = String::from("hello");

let r1 = &mut s;
let r2 = &mut s; // зј–иҜ‘й”ҷиҜҜпјҒ

println!("{}, {}", r1, r2);

и§ЈеҶіж–№жі•йҖҡеёёжҳҜзј©е°Ҹеј•з”Ёзҡ„дҪңз”Ёеҹҹпјҡ

let mut s = String::from("hello");

{
    let r1 = &mut s;
    println!("{}", r1);
} // r1 еңЁиҝҷйҮҢзҰ»ејҖдҪңз”Ёеҹҹ

let r2 = &mut s; // зҺ°еңЁ OK дәҶпјҒ
println!("{}", r2);

еҸҰдёҖдёӘеёёи§Ғй”ҷиҜҜжҳҜиҜ•еӣҫеңЁжҢҒжңүеј•з”Ёзҡ„еҗҢж—¶дҝ®ж”№еҺҹе§Ӣж•°жҚ®пјҡ

let s = String::from("hello");

let r = &s;

s.push_str(", world!"); // й”ҷиҜҜпјҒдёҚиғҪеңЁжңүдёҚеҸҜеҸҳеј•з”Ёж—¶дҝ®ж”№

println!("{}", r);

еҠЁжүӢе®һи·ө

зҺ°еңЁиҪ®еҲ°дҪ дәҶпјҒе°қиҜ•и§ЈеҶід»ҘдёӢз»ғд№ пјҡ

  1. жүҖжңүжқғз»ғд№ пјҡзј–еҶҷдёҖдёӘеҮҪж•°пјҢжҺҘж”¶дёҖдёӘ Vec<i32>пјҢеҗ‘е…¶дёӯж·»еҠ дёҖдёӘе…ғзҙ пјҢз„¶еҗҺиҝ”еӣһдҝ®ж”№еҗҺзҡ„еҗ‘йҮҸгҖӮжҖқиҖғдёәд»Җд№ҲйңҖиҰҒиҝ”еӣһеҗ‘йҮҸгҖӮ

  2. еҖҹз”Ёз»ғд№ пјҡзј–еҶҷдёҖдёӘеҮҪж•°пјҢи®Ўз®—еҗ‘йҮҸдёӯжүҖжңүе…ғзҙ зҡ„е’ҢпјҢдҪҶдёҚиҰҒиҺ·еҸ–еҗ‘йҮҸзҡ„жүҖжңүжқғгҖӮдҪҝз”Ёеј•з”ЁжқҘе®һзҺ°гҖӮ

  3. еӯ—з¬ҰдёІеӨ„зҗҶпјҡзј–еҶҷдёҖдёӘеҮҪж•°пјҢжҺҘж”¶дёҖдёӘеӯ—з¬ҰдёІеҲҮзүҮпјҢиҝ”еӣһе…¶дёӯжңҖй•ҝзҡ„еҚ•иҜҚгҖӮзЎ®дҝқжӯЈзЎ®еӨ„зҗҶз”ҹе‘Ҫе‘ЁжңҹгҖӮ

  4. и°ғиҜ•з»ғд№ пјҡдёӢйқўзҡ„д»Јз Ғжңүд»Җд№Ҳй—®йўҳпјҹеҰӮдҪ•дҝ®еӨҚпјҹ

    fn main() {
    let s1 = String::from("hello");
    let s2 = s1;
    println!("{}", s1);
}

и®°дҪҸпјҡRust зҡ„зј–иҜ‘еҷЁжҳҜдҪ жңҖеҘҪзҡ„жңӢеҸӢпјҒе®ғдјҡз»ҷдҪ жё…жҷ°зҡ„й”ҷиҜҜдҝЎжҒҜпјҢе‘ҠиҜүдҪ е“ӘйҮҢиҝқеҸҚдәҶжүҖжңүжқғи§„еҲҷгҖӮ

е°Ҹз»“дёҺжҖқиҖғ

Rust зҡ„жүҖжңүжқғзі»з»ҹеҸҜиғҪеңЁдёҖејҖе§ӢзңӢиө·жқҘеҫҲдёҘж јпјҢдҪҶе®ғжҸҗдҫӣдәҶе·ЁеӨ§зҡ„еҘҪеӨ„пјҡ

  • еҶ…еӯҳе®үе…ЁпјҡеңЁзј–иҜ‘ж—¶е°ұйҳІжӯўдәҶеҶ…еӯҳй”ҷиҜҜ
  • йӣ¶жҲҗжң¬жҠҪиұЎпјҡдёҚйңҖиҰҒеһғеңҫеӣһж”¶еҷЁпјҢиҝҗиЎҢж—¶ејҖй”Җдёәйӣ¶
  • е№¶еҸ‘е®үе…ЁпјҡжүҖжңүжқғи§„еҲҷеӨ©з„¶йҳІжӯўдәҶж•°жҚ®з«һдәү

еҪ“дҪ жҺҢжҸЎдәҶжүҖжңүжқғе’ҢеҖҹз”Ёзҡ„жҰӮеҝөеҗҺпјҢдҪ дјҡеҸ‘зҺ°е®ғд»¬е®һйҷ…дёҠи®©зј–зЁӢеҸҳеҫ—жӣҙеҠ з®ҖеҚ•е’Ңе®үе…ЁгҖӮдҪ дёҚеҶҚйңҖиҰҒжӢ…еҝғеҶ…еӯҳжі„жјҸгҖҒз©әжҢҮй’ҲжҲ–ж•°жҚ®з«һдәүвҖ”вҖ”зј–иҜ‘еҷЁе·Із»ҸдёәдҪ еӨ„зҗҶдәҶиҝҷдәӣй—®йўҳпјҒ

жҖқиҖғйўҳпјҡ

  1. дёәд»Җд№Ҳ Rust йңҖиҰҒеҢәеҲҶз§»еҠЁиҜӯд№үе’ҢеӨҚеҲ¶иҜӯд№үпјҹ
  2. еңЁд»Җд№Ҳжғ…еҶөдёӢдҪ дјҡйҖүжӢ©дҪҝз”Ёеј•з”ЁиҖҢдёҚжҳҜиҺ·еҸ–жүҖжңүжқғпјҹ
